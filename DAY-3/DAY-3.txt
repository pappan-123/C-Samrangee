Draw the FlowChart:

1.Find the largest among three different numbers entered by the user.
2.Find all the roots of a quadratic equation ax2+bx+c=0
3.Draw a flowchart to check whether a student has passed or failed based on marks. If the marks are greater than or equal to 40, display ‘Passed’; otherwise, display ‘Failed’


Write the Psedo Code:

1.
#include <stdio.h>
int main() {
int n, f = 1, i;
// Ask for user input
printf("Please enter a non-negative number: ");
scanf("%d", &n);
// Check if the input is negative
if (n < 0) {
printf("Error: Factorial is not defined for negative numbers.\n");
} else {
// Calculate factorial
for (i = 1; i <= n; i++) {
    f = f * i;
}
// Display the result
printf("The factorial of %d is %d\n", n, f);
}
return 0;
}

2.
#include <stdio.h>
int main() {
int number;
// Ask for user input
printf("Enter a number: ");
scanf("%d", &number);
// Use ternary operator to check if the number is even or odd
(number % 2 == 0) ? printf("The number is even\n") : printf("The number is odd\n");
return 0;
}

3.
#include <stdio.h>

int main() {
    int marks;

    // Take input from user
    printf("Enter your marks: ");
    scanf("%d", &marks);

    // Condition check
    if (marks >= 40) {
        printf("Passed");
    } else {
        printf("Failed");
    }

    return 0;
}
